#include "Nodo.h"

Nodo::Nodo(){
    this->valor = 0;
    this->next = nullptr;
}

Nodo::Nodo(int v){
    this->valor = v;
    this->next = nullptr;
}

Nodo::Nodo(int v, Nodo* n){
    this->valor = v;
    this->next = n;
}

int Nodo::getValor(){
    return this->valor;
}

Nodo* Nodo::getNext(){
    return this->next;
}

void Nodo::setValor(int v){
    this->valor = v;
}

void Nodo::setNext(Nodo* n){
    this->next = n;
}

Nodo::~Nodo(){
    delete this->next;
}
